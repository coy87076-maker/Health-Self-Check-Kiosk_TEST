function doPost(e) {
  const sheet = SpreadsheetApp.openById('1oOnGPe1KcdeNvi5BVzTz8QSoBonZE2iZt3ezyEnTZnc').getSheetByName('HealthCheck');
  const data = e.parameter;

  // Auto-add header row if sheet is empty
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['Date', 'Name', 'Age', 'Sex', 'Weight (kg)', 'Height (cm)', 'BMI', 'Category']);
  }

  sheet.appendRow([
    new Date(),
    data.name,
    data.age,
    data.sex,
    data.weight,
    data.heightCm,
    data.bmi,
    data.category
  ]);

  return ContentService.createTextOutput('Success');
}

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({ status: 'ok', message: 'Health Kiosk API is running.' }))
    .setMimeType(ContentService.MimeType.JSON);
}
